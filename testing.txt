# Header to init object
HEADER 0
:START
class Thing:
	def __init__(self, name, age, fav_thing):
		self.name = name
		self.age = age
		self.fav_thing = fav_thing
	
	def getName(self):
		return self.name
		
	def getAge(self):
		return self.age

	def getFav(self):
		return self.fav_thing

thing = Thing("Name", 13, "sports")

:END

# First template
TEMPLATE SIMPLEOP THING OP VALUE
:START
# Testing $THING$
if ($THING$ $OP$ $VALUE$):
	print("PASS")
else:
	print("FAIL $THING$ !$OP$ $VALUE$")
	exit(0)
:END
